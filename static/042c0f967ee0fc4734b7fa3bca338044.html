2017-12-22 [shell]

登陆服务器每次都输入密码很不舒服，所以总结了一个小脚本。

#!/bin/bash
USER=activemq
current_node=192.168.1.45
PASSWD=activemq
/usr/bin/expect <<EOF
    spawn ssh $USER@$current_node
    expect "*(yes/no)?*" {
        send -- "yes\r"
        expect "*?assword:*"
        send -- "$PASSWD\r"
    }  "*?assword:*"
        send -- "$PASSWD\r"
        expect "*# "
    send -- "exit \r"
EOF
