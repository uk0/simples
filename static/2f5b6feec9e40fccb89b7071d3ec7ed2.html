  • 监听麦克风进行识别歌曲
#!/usr/bin/python  
# -*- coding: utf-8 -*-  
  
  
from dejavu import Dejavu  
from dejavu.recognize import FileRecognizer, MicrophoneRecognizer  
  
config = {  
    "database": {  
        "host": "127.0.0.1",  
        "user": "root",  
        "passwd": "passw0rd",  
        "db": "dejavu",  
    },  
    "database_type": "mysql"  
}  
djv = Dejavu(config)  
  
# 识别从mic输入的音频 只获得5秒的音频  
secs = 5  
song = djv.recognize(MicrophoneRecognizer, seconds=secs)  
if song is None:  
    print "No Match"  
else:  
    print(song)
